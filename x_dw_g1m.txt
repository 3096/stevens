NUNV FORMAT

h01 = uint32 magic     - 4E 55 4E 56
h02 = uint32 type      - 30 30 31 30 or 30 30 31 31
h03 = uint32 chunksize - 00 00 0C 38
h04 = uint32 n_schunks - 00 00 00 01

00 05 00 01 ITEM
----------------

00 05 00 01 - chunk id
00 00 0C 28 - chunksize
00 00 00 02 - number of subchunks


h01 = uint32 - 80 00 00 0A
h02 = uint32 - 00 00 00 1B (number of control points)
h03 = uint32 - 00 00 00 03
h04 = uint32 - 00 00 00 02
h05 = uint32 - 00 00 00 01
h06 = real32 - 40 33 33 33
h07 = real32[3] - 3E E6 66 66 3D CC CC CD 3E 80 00 00
h08 = real32[3] - 3F 19 99 9A 3F 80 00 00 3D CC CC CD

h12 = real32[3] - 3C 88 89 3B 00 00 00 00 3F 80 00 00 00
h13 = uint32    - 00 01 01 00
